-- AIVA Dashboard v12.2: полный сброс битого состояния городов
-- Supabase → SQL Editor → New query → Run
-- Это очистит сохранённый сломанный state по городам и дашборд заново создаст рабочую структуру.

delete from public.dashboard_state
where id in ('main_astana', 'main_almaty');

insert into public.dashboard_state (id, data)
values
  ('main_astana', '{}'::jsonb),
  ('main_almaty', '{}'::jsonb)
on conflict (id) do update
set data = '{}'::jsonb,
    updated_at = now();
